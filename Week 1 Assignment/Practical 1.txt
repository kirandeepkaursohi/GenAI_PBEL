for i in range(5):

    roll_no = input("Enter Roll Number: ")

        marks.append(mark)

    students_details[roll_no] = {"name": name, "marks": marks}


def total_marks(marks):
    return sum(marks)


def percentage(total_marks, marks):
    percentage = (total_marks / len(marks))
    return percentage


grade_criteria = ("A", "B", "C", "D", "F")


def grade(percentage):
    if percentage >= 90:
        grade = "A"
    elif percentage >= 75:
        grade = "B"
    elif percentage >= 60:
        grade = "C"
    elif percentage >= 40:
        grade = "D"
    else:
        grade = "F"
    return grade
